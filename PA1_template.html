dt2 <- read.csv("activity.csv")

# Replacing NAs with the mean number of steps of the interval for all days.
for (x in 1:length(dt2[, 1])) {
    interval <- dt2[x, 3]
    if (is.na(dt2[x, 1])) {
        interval_mean <- dt_mean[dt_mean$interval == interval, ][2]
        dt2[x, 1] <- interval_mean
    }
}
